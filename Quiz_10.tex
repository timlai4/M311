\question Set up but do not evaluate the double integrals to compute the areas of the following:
\begin{parts}
\part[10] The unit semicircle above the $x$-axis. 
\begin{solution}
\[
\int_{-1}^1 \int_0^{\sqrt{1 - x^2}} 1 \, \ dy dx
\]
\end{solution}
\part[10] The region bounded by $x = y ^2$ and $y = x-2$. 
\begin{solution}
\[
\int_1^4 \int_{y^2}^{y+2} 1 \, \ dx dy
\]
\end{solution}
\end{parts}
\question[20] Compute 
\[
\int_0^1 \int_y^1 e^{-x^2} \, dx dy
\]
Here is a guided approach that you may optionally follow.
\begin{parts}
\part[5] Draw and shade the region
\part[5] Exchange the integrals with the aid of the picture drawn in the previous part. 
\part[10] Compute the iterated integrals $\iint _R e^{-x^2}\, dy dx$. 
\end{parts}
\begin{solution}
\[
\int_0^1 \int_y^1 e^{-x^2} \, \ dx dy = \int_0^1 \int_0^x  e^{-x^2} \, \ dy dx = \int_0^1 x e^{-x^2} dx
\]
Then use the $u$-sub $u = -x^2$. 
\end{solution}
\question[20] For the region $R$ bounded by the curves $x = y^2$ and $y = x$, compute
\[
 \iint_R 3x + 2y \, dA
\]
\begin{solution}
\[
 \iint_R 3x + 2y \, dA = \int_0^1 \int_{y^2}^y 3x + 2y \, \ dx dy
\]
or
\[
= \int_0^1 \int_{x}^{\sqrt{x}} 3x + 2y \, \ dy dx
\]
\end{solution}
